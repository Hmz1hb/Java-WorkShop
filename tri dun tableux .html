Tri d'un tableau

Enoncé

Ecrire un programme Javascript qui permet de trier un tableau d’entier (Utiliser n’importe quel algorithme de tri mais surtout pas la fonction Sort de l’objet ary1). L’utilisateur commence tout d’abord par saisir la taille du tableau, ensuite il remplit le tableau pour cliquer à la fin sur un bouton trier : celui-ci affichera le tableau en ordre croissant ou décroissant.

function InpChek(){
var inputs = document.getElementById('inp')
var boxNum = parseInt(document.getElementById('box-num').value)
inputs.innerHTML="";
for(x=1 ; x<= boxNum ;x++){
var inp = document.createElement('INPUT');
inp.setAttribute('id', 'inp'+x)
inputs.appendChild(inp)
}
}
function tri()
{
var boxNum = parseInt(document.getElementById('box-num').value)
var outinpt = document.getElementById('out-inp')
outinpt.innerHTML="";
for(x=0 ; x< boxNum ;x++)
{
var inpsr = document.createElement('INPUT')
inpsr.setAttribute('id', 'out-inp'+x)
outinpt.appendChild(inpsr)
}
const ary1 = []
const ary2 = []
for (let i = 1; i <= boxNum; i++)
{
ary1.push(parseInt(document.getElementById('inp'+i).value))
}
for (let i = 0; i < boxNum; i++)
{
let min = Math.min(...ary1)
const index = ary1.indexOf(min);
ary2.push(min)
document.getElementById('out-inp'+i).value = min
ary1.splice(index,1)
}
}
