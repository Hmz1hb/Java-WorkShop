Valider Une Date

Enoncé

Ecrire un script qui permet de saisir les trois champs d’une date (day, mois et année) et d’indiquer si la date est correcte ou non

var day = document.getElementById('day');
var month = document.getElementById('month');
var year = document.getElementById('year');
document.getElementById('val').onclick=checkDate;
var validDtae = document.getElementById('out');

function checkDate() {
if ((year.value > 1900 && year.value < 2025) && day.value >= 1 &&
(((month.value == 4 || month.value == 6 || month.value == 9 || month.value == 11) && day.value <= 30) ||
((month.value == 1 || month.value == 3 || month.value == 5 || month.value == 7 || month.value == 8 || month.value == 10 || month.value == 12) && day.value <= 31) ||
(month.value == 2 && day.value <= 28))) {
validDtae.innerHTML = `VALIDE`
} else {
validDtae.innerHTML = `Cette date est Unvalide`
} }
